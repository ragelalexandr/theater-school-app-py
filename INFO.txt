theatre_platform/                # Корневой каталог проекта
│
├── accounts/                    # Приложение accounts — содержит всю бизнес-логику, связанную с пользователями, курсами, отзывами и дополнительным административным функционалом
│   ├── __init__.py
│   ├── admin.py               # Регистрация моделей в стандартной админке Django
│   ├── admin_views.py         # Представления для кастомного административного раздела (CRUD для представлений, курсов, списков пользователей, модерация отзывов)
│   ├── apps.py                # Конфигурация приложения accounts (например, AccountsConfig)
│   ├── decorators.py          # Дополнительные декораторы (например, для проверки, что пользователь является преподавателем)
│   ├── forms.py               # Формы для создания/редактирования моделей (в т.ч. для Performance и Course)
│   ├── migrations/            # Миграции базы данных для приложения
│   ├── models.py              # Модели: CustomUser, Profile, Performance, Course, Enrollment, Portfolio, Review
│   ├── signals.py             # Сигналы (например, для автоматического создания профиля после создания пользователя)
│   ├── teacher_dashboard.html # Шаблон страницы панели преподавателя (если используется)
│   ├── tests.py               # Тесты для приложения accounts
│   ├── urls.py                # Публичные маршруты приложения accounts (например, вход, регистрация, домашняя страница конкретного раздела)
│   └── urls_admin.py          # Маршруты для кастомного административного раздела (с путями для управления представлениями, курсами, пользователями, отзывами)
│   └── views.py               # Публичные представления (например, функция home, регистрации, авторизации и т.п.)
│
├── static/                      # Статические файлы проекта: css, изображения, js, шрифты и др.
│   ├── css/                   # Стили (включая файлы шрифтов и svg-иконки)
│   ├── images/                # Изображения для блоков сайта (about, courses, header, performances и т.д.)
│   └── js/                    # Скрипты JavaScript
│   └── favicon.ico            # Иконка сайта
│
├── templates/                   # Шаблоны для всего сайта
│   ├── admin/                 # Шаблоны кастомного административного раздела (dashboard.html, performances_list.html, performance_form.html, course_list.html и др.)
│   ├── blocks/                # Частичные шаблоны (блоки about, contacts, courses, header, performers, teachers, team, tour, why_choose_us и т.д.)
│   ├── registration/          # Шаблоны для регистрации и авторизации (login.html, registration.html)
│   ├── base.html              # Базовый шаблон, от которого наследуются все страницы сайта (с включением общего header, footer, меню)
│   ├── history.html           # Шаблон для страницы истории (если используется)
│   └── home.html              # Главная страница сайта
│
├── theatre_platform/            # Основной пакет проекта (настройки, URL-конфигурация, WSGI/ASGI)
│   ├── __init__.py
│   ├── asgi.py
│   ├── settings.py            # Основные настройки проекта (включая INSTALLED_APPS, TEMPLATES, STATICFILES_DIRS)
│   ├── urls.py                # Основные маршруты проекта: подключение стандартной админки (admin/) и маршрутов публичной или административной части (например, '' и 'admin-panel/')
│   └── wsgi.py
│
├── db.sqlite3                   # Файл базы данных (SQLite по умолчанию)
│
├── manage.py                    # Скрипт управления проектом Django (запуск сервера, миграции, shell и т.д.)
│
├── requirements.txt             # Файл с зависимостями проекта, который можно обновить через `pip freeze > requirements.txt`
│
└── TASK.txt                     # Файл с описанием задач (план работы, комментарии к реализации и т.д.)
